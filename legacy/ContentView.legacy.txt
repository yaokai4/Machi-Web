private extension Int {
    var compactCountText: String {
        let value = Double(self)
        if self < 1_000 {
            return "\(self)"
        } else if self < 10_000 {
            let compact = value / 1_000
            return compact.truncatingRemainder(dividingBy: 1) == 0
                ? "\(Int(compact))K"
                : String(format: "%.1fK", compact)
        } else if self < 1_000_000 {
            return "\(self / 1_000)K"
        } else {
            let compact = value / 1_000_000
            return compact.truncatingRemainder(dividingBy: 1) == 0
                ? "\(Int(compact))M"
                : String(format: "%.1fM", compact)
        }
    }
}

private extension CMTime {
    var composeDurationText: String {
        let secondsValue = CMTimeGetSeconds(self)
        let totalSeconds = secondsValue.isFinite ? max(0, Int(secondsValue.rounded())) : 30
        let minutes = totalSeconds / 60
        let seconds = totalSeconds % 60
        return String(format: "%02d:%02d", minutes, seconds)
    }
}

private enum SharedBackend {
    static let baseURL = URL(string: "http://127.0.0.1:8787")!

    static func bootstrap(userID: String) async -> SharedBackendSnapshot? {
        var components = URLComponents(url: baseURL.appending(path: "/api/bootstrap"), resolvingAgainstBaseURL: false)
        components?.queryItems = [URLQueryItem(name: "user_id", value: userID)]
        guard let url = components?.url else { return nil }
        var request = URLRequest(url: url)
        request.timeoutInterval = 4
        guard let (data, response) = try? await URLSession.shared.data(for: request),
              let http = response as? HTTPURLResponse,
              (200..<300).contains(http.statusCode)
        else {
            return nil
        }
        return try? JSONDecoder.sharedBackend.decode(SharedBackendSnapshot.self, from: data)
    }

    static func createPost(userID: String, content: String, repostOfID: String? = nil) async {
        await post("/api/posts", body: ["user_id": userID, "content": content, "repost_of_id": repostOfID as Any])
    }

    static func togglePostInteraction(postID: String, userID: String, kind: InteractionKind) async {
        await post("/api/posts/\(postID)/interaction", body: ["user_id": userID, "kind": kind.rawValue])
    }

    static func createComment(postID: String, userID: String, content: String, replyToUserID: String?) async {
        await post("/api/posts/\(postID)/comments", body: ["user_id": userID, "content": content, "reply_to_user_id": replyToUserID as Any])
    }

    static func toggleCommentLike(commentID: String, userID: String) async {
        await post("/api/comments/\(commentID)/like", body: ["user_id": userID])
    }

    static func toggleFollow(followerID: String, followingID: String) async {
        await post("/api/follow", body: ["follower_id": followerID, "following_id": followingID])
    }

    static func createMessage(senderID: String, receiverID: String, content: String, attachmentSymbol: String? = nil) async {
        await post("/api/messages", body: ["sender_id": senderID, "receiver_id": receiverID, "content": content, "attachment_symbol": attachmentSymbol as Any])
    }

    static func updateUser(_ user: AppUser) async {
        await patch("/api/users/\(user.id)", body: [
            "handle": user.handle,
            "display_name": user.displayName,
            "password": user.password,
            "bio": user.bio,
            "location": user.location,
            "avatar_symbol": user.avatarSymbol,
            "avatar_color": user.avatarColorName,
            "email": user.accountEmail,
            "membership_tier": user.membershipTierRaw,
            "is_verified": user.isVerified
        ])
    }

    private static func post(_ path: String, body: [String: Any]) async {
        await request(path, method: "POST", body: body)
    }

    private static func patch(_ path: String, body: [String: Any]) async {
        await request(path, method: "PATCH", body: body)
    }

    private static func request(_ path: String, method: String, body: [String: Any]) async {
        var request = URLRequest(url: baseURL.appending(path: path))
        request.httpMethod = method
        request.timeoutInterval = 4
        request.setValue("application/json", forHTTPHeaderField: "Content-Type")
        let cleanBody = body.compactMapValues { value -> Any? in
            if let optional = value as? OptionalValue, optional.isNil { return nil }
            return value
        }
        request.httpBody = try? JSONSerialization.data(withJSONObject: cleanBody)
        _ = try? await URLSession.shared.data(for: request)
    }
}

private extension JSONDecoder {
    static var sharedBackend: JSONDecoder {
        let decoder = JSONDecoder()
        let formatter = ISO8601DateFormatter()
        formatter.formatOptions = [.withInternetDateTime, .withFractionalSeconds]
        decoder.dateDecodingStrategy = .custom { decoder in
            let container = try decoder.singleValueContainer()
            let value = try container.decode(String.self)
            if let date = formatter.date(from: value) {
                return date
            }
            let fallback = ISO8601DateFormatter()
            if let date = fallback.date(from: value) {
                return date
            }
            throw DecodingError.dataCorruptedError(in: container, debugDescription: "Invalid ISO8601 date: \(value)")
        }
        return decoder
    }
}

private enum AppLanguage: String, CaseIterable, Identifiable {
    case system
    case zh
    case ja
    case en

    var id: String { rawValue }

    static func resolved(from code: String) -> AppLanguage {
        if let selected = AppLanguage(rawValue: code), selected != .system {
            return selected
        }

        let preferred = Locale.preferredLanguages.first?.lowercased() ?? "zh"
        if preferred.hasPrefix("ja") { return .ja }
        if preferred.hasPrefix("en") { return .en }
        return .zh
    }

    var displayName: String {
        switch self {
        case .system: "跟随系统 / System"
        case .zh: "中文"
        case .ja: "日本語"
        case .en: "English"
        }
    }

    func displayName(_ language: AppLanguage) -> String {
        switch self {
        case .system:
            switch language {
            case .zh: "跟随系统"
            case .ja: "システムに従う"
            case .en: "Follow System"
            case .system: "Follow System"
            }
        case .zh: "中文"
        case .ja: "日本語"
        case .en: "English"
        }
    }
}

private struct AppLanguageKey: EnvironmentKey {
    static let defaultValue: AppLanguage = .zh
}

private extension EnvironmentValues {
    var appLanguage: AppLanguage {
        get { self[AppLanguageKey.self] }
        set { self[AppLanguageKey.self] = newValue }
    }
}

@Model
final class AppUser {
    @Attribute(.unique) var id: String
    @Attribute(.unique) var handle: String
    var displayName: String
    var password: String
    var bio: String
    var location: String
    var avatarSymbol: String
    var avatarColorName: String
    @Attribute(.externalStorage) var avatarImageData: Data?
    @Attribute(.externalStorage) var coverImageData: Data?
    var joinedAt: Date
    var accountEmail: String = ""
    var membershipTierRaw: String = MembershipTier.free.rawValue
    var isVerified: Bool = false
    var serverRegion: String = "ap-northeast-1"
    var serverSyncEnabled: Bool = false
    var websiteURL: String = ""

    init(
        id: String = UUID().uuidString,
        handle: String,
        displayName: String,
        password: String,
        bio: String = "",
        location: String = "",
        avatarSymbol: String = "person.fill",
        avatarColorName: String = "black",
        avatarImageData: Data? = nil,
        coverImageData: Data? = nil,
        joinedAt: Date = .now,
        accountEmail: String = "",
        membershipTier: MembershipTier = .free,
        isVerified: Bool = false,
        serverRegion: String = "ap-northeast-1",
        serverSyncEnabled: Bool = false,
        websiteURL: String = ""
    ) {
        self.id = id
        self.handle = handle.normalizedHandle
        self.displayName = displayName
        self.password = password
        self.bio = bio
        self.location = location
        self.avatarSymbol = avatarSymbol
        self.avatarColorName = avatarColorName
        self.avatarImageData = avatarImageData
        self.coverImageData = coverImageData
        self.joinedAt = joinedAt
        self.accountEmail = accountEmail
        self.membershipTierRaw = membershipTier.rawValue
        self.isVerified = isVerified
        self.serverRegion = serverRegion
        self.serverSyncEnabled = serverSyncEnabled
        self.websiteURL = websiteURL
    }
}

@Model
final class FeedPost {
    @Attribute(.unique) var id: String
    var authorID: String
    var content: String
    var createdAt: Date
    var repostOfID: String?
    var viewCount: Int
    var mediaKindRaw: String = PostMediaKind.none.rawValue
    var mediaSymbol: String = ""
    var mediaColorName: String = "blue"
    var mediaTitle: String = ""
    var videoDuration: String = ""
    @Attribute(.externalStorage) var mediaImageData1: Data?
    @Attribute(.externalStorage) var mediaImageData2: Data?
    @Attribute(.externalStorage) var mediaImageData3: Data?
    @Attribute(.externalStorage) var mediaImageData4: Data?
    @Attribute(.externalStorage) var videoData: Data?

    init(
        id: String = UUID().uuidString,
        authorID: String,
        content: String,
        createdAt: Date = .now,
        repostOfID: String? = nil,
        viewCount: Int = 0,
        mediaKind: PostMediaKind = .none,
        mediaSymbol: String = "",
        mediaColorName: String = "blue",
        mediaTitle: String = "",
        videoDuration: String = "",
        mediaImageData: [Data] = [],
        videoData: Data? = nil
    ) {
        self.id = id
        self.authorID = authorID
        self.content = content
        self.createdAt = createdAt
        self.repostOfID = repostOfID
        self.viewCount = viewCount
        self.mediaKindRaw = mediaKind.rawValue
        self.mediaSymbol = mediaSymbol
        self.mediaColorName = mediaColorName
        self.mediaTitle = mediaTitle
        self.videoDuration = videoDuration
        self.mediaImageData1 = mediaImageData.indices.contains(0) ? mediaImageData[0] : nil
        self.mediaImageData2 = mediaImageData.indices.contains(1) ? mediaImageData[1] : nil
        self.mediaImageData3 = mediaImageData.indices.contains(2) ? mediaImageData[2] : nil
        self.mediaImageData4 = mediaImageData.indices.contains(3) ? mediaImageData[3] : nil
        self.videoData = videoData
    }
}

    }

    @Query(sort: \AppUser.joinedAt) private var users: [AppUser]
    @Query(sort: \FeedPost.createdAt) private var posts: [FeedPost]
    @Query private var follows: [FollowRelation]
    @Query private var interactions: [PostInteraction]
    @Query private var comments: [PostComment]
    @Query private var messages: [DirectMessage]
    @State private var selectedTab = AppTab.home
    @State private var isShowingComposer = false

    private var currentUser: AppUser? {
        users.first { $0.id == currentUserID }
    }

    var body: some View {
        Group {
            if let currentUser {
                MainAppView(
                    currentUser: currentUser,
                    selectedTab: $selectedTab,
                    isShowingComposer: $isShowingComposer
                )
            } else {
                AuthView(currentUserID: $currentUserID)
            }
        }
        .environment(\.appLanguage, AppLanguage.resolved(from: appLanguageCode))
        .task(id: currentUserID) {
            seedDatabaseIfNeeded()
            await syncFromSharedBackendIfAvailable()
        }
    }

    @MainActor
    private func syncFromSharedBackendIfAvailable() async {
        let userID = currentUserID.isEmpty ? users.first(where: { $0.handle == "kaizi" })?.id : currentUserID
        guard let userID, let snapshot = await SharedBackend.bootstrap(userID: userID) else { return }
        upsert(snapshot)
        try? modelContext.save()
    }

    @MainActor
    private func upsert(_ snapshot: SharedBackendSnapshot) {
        for remote in snapshot.users {
            if let user = users.first(where: { $0.id == remote.id }) ?? users.first(where: { $0.handle == remote.handle }) {
                user.handle = remote.handle
                user.displayName = remote.displayName
                user.password = remote.password
                user.bio = remote.bio
                user.location = remote.location
                user.avatarSymbol = remote.avatarSymbol
                user.avatarColorName = remote.avatarColor
                user.joinedAt = remote.joinedAt
                user.accountEmail = remote.email
                user.membershipTierRaw = remote.membershipTier
                user.isVerified = remote.isVerified != 0
                user.serverSyncEnabled = true
            } else {
                modelContext.insert(AppUser(
                    id: remote.id,
                    handle: remote.handle,
                    displayName: remote.displayName,
                    password: remote.password,
                    bio: remote.bio,
                    location: remote.location,
                    avatarSymbol: remote.avatarSymbol,
                    avatarColorName: remote.avatarColor,
                    joinedAt: remote.joinedAt,
                    accountEmail: remote.email,
                    membershipTier: MembershipTier(rawValue: remote.membershipTier) ?? .free,
                    isVerified: remote.isVerified != 0,
                    serverSyncEnabled: true
                ))
            }
        }

        for remote in snapshot.posts {
            if let post = posts.first(where: { $0.id == remote.id }) {
                post.authorID = remote.authorID
                post.content = remote.content
                post.createdAt = remote.createdAt
                post.repostOfID = remote.repostOfID
                post.viewCount = remote.viewCount
            } else {
                modelContext.insert(FeedPost(
                    id: remote.id,
                    authorID: remote.authorID,
                    content: remote.content,
                    createdAt: remote.createdAt,
                    repostOfID: remote.repostOfID,
                    viewCount: remote.viewCount
                ))
            }
        }

        for remote in snapshot.comments {
            if let comment = comments.first(where: { $0.id == remote.id }) {
                comment.postID = remote.postID
                comment.authorID = remote.authorID
                comment.content = remote.content
                comment.createdAt = remote.createdAt
                comment.replyToUserID = remote.replyToUserID
            } else {
                modelContext.insert(PostComment(
                    id: remote.id,
                    postID: remote.postID,
                    authorID: remote.authorID,
                    content: remote.content,
                    createdAt: remote.createdAt,
                    replyToUserID: remote.replyToUserID
                ))
            }
        }

        for remote in snapshot.interactions {
            guard let kind = InteractionKind(rawValue: remote.kind) else { continue }
            let existing = interactions.first {
                $0.id == remote.id || ($0.postID == remote.targetID && $0.userID == remote.userID && $0.kind == remote.kind)
            }
            if let interaction = existing {
                interaction.postID = remote.targetID
                interaction.userID = remote.userID
                interaction.kind = remote.kind
                interaction.createdAt = remote.createdAt
            } else {
                modelContext.insert(PostInteraction(
                    id: remote.id,
                    postID: remote.targetID,
                    userID: remote.userID,
                    kind: kind,
                    createdAt: remote.createdAt
                ))
            }
        }

        for remote in snapshot.follows {
            let existing = follows.first {
                $0.id == remote.id || ($0.followerID == remote.followerID && $0.followingID == remote.followingID)
            }
            if let follow = existing {
                follow.followerID = remote.followerID
                follow.followingID = remote.followingID
                follow.createdAt = remote.createdAt
            } else {
                modelContext.insert(FollowRelation(
                    id: remote.id,
                    followerID: remote.followerID,
                    followingID: remote.followingID,
                    createdAt: remote.createdAt
                ))
            }
        }

        for remote in snapshot.messages {
            if let message = messages.first(where: { $0.id == remote.id }) {
                message.senderID = remote.senderID
                message.receiverID = remote.receiverID
                message.content = remote.content
                message.createdAt = remote.createdAt
                message.isRead = remote.isRead != 0
                message.attachmentSymbol = remote.attachmentSymbol
            } else {
                modelContext.insert(DirectMessage(
                    id: remote.id,
                    senderID: remote.senderID,
                    receiverID: remote.receiverID,
                    content: remote.content,
                    createdAt: remote.createdAt,
                    isRead: remote.isRead != 0,
                    attachmentSymbol: remote.attachmentSymbol
                ))
            }
        }
    }

    private func seedFollowPairs(_ pairs: [(AppUser, AppUser)]) {
        for (follower, following) in pairs where follower.id != following.id {
            if follows.contains(where: { $0.followerID == follower.id && $0.followingID == following.id }) {
                continue
            }
            modelContext.insert(FollowRelation(followerID: follower.id, followingID: following.id))
        }

private struct AuthView: View {
    @Environment(\.appLanguage) private var appLanguage
    @Environment(\.modelContext) private var modelContext
    @Binding var currentUserID: String
    @Query(sort: \AppUser.joinedAt) private var users: [AppUser]
    @State private var mode = AuthMode.login
    @State private var handle = ""
    @State private var displayName = ""
    @State private var password = ""
    @State private var message = ""
    @State private var isPasswordVisible = false

    private var normalizedHandle: String {
        handle.normalizedHandle
    }

    private var isFormReady: Bool {
        if mode == .register {
            return !normalizedHandle.isEmpty
            && !displayName.trimmingCharacters(in: .whitespacesAndNewlines).isEmpty
            && password.count >= 6
        }

        return !normalizedHandle.isEmpty && !password.isEmpty
    }

    var body: some View {
        ZStack {
            LinearGradient(
                colors: [Color(.systemBackground), Color(.secondarySystemBackground)],
                startPoint: .top,
                endPoint: .bottom
            )
            .ignoresSafeArea()

            ScrollView {
                VStack(spacing: 28) {
                    AuthHeader(mode: mode)

                    VStack(spacing: 18) {
                        Picker(L("login", appLanguage), selection: $mode) {
                            ForEach(AuthMode.allCases) { mode in
                                Text(mode.title(appLanguage)).tag(mode)
                            }
                        }
                        .pickerStyle(.segmented)
                        .padding(4)
                        .background(Color(.tertiarySystemBackground))
                        .clipShape(RoundedRectangle(cornerRadius: 14))

                        VStack(spacing: 12) {
                            AuthInputField(
                                title: L("username", appLanguage),
                                placeholder: "kaizi",
                                text: $handle,
                                icon: "at",
                                textContentType: .username,
                                keyboardType: .asciiCapable,
                                autocapitalization: .never
                            )

                            if mode == .register {
                                AuthInputField(
                                    title: L("displayName", appLanguage),
                                    placeholder: L("nicknamePlaceholder", appLanguage),
                                    text: $displayName,
                                    icon: "person.text.rectangle",
                                    textContentType: .name
                                )
                                .transition(.opacity.combined(with: .move(edge: .top)))
                            }

                            AuthPasswordField(
                                password: $password,
                                isVisible: $isPasswordVisible,
                                mode: mode
                            )
                        }

                        if mode == .register {
                            PasswordStrengthView(password: password)
                        }

                        if !message.isEmpty {
                            Label(message, systemImage: "exclamationmark.circle.fill")
                                .font(.footnote.weight(.medium))
                                .foregroundStyle(.red)
                                .frame(maxWidth: .infinity, alignment: .leading)
                        }

                        Button {
                            submit()
                        } label: {
                            HStack {
                                Text(mode == .login ? L("loginKaiX", appLanguage) : L("createAccount", appLanguage))
                                Image(systemName: "arrow.right")
                            }
                            .font(.headline)
                            .frame(maxWidth: .infinity)
                            .frame(height: 52)
                            .foregroundStyle(.white)
                            .background(isFormReady ? Color.black : Color.gray)
                            .clipShape(RoundedRectangle(cornerRadius: 15))
                        }
                        .disabled(!isFormReady)

                        Button {
                            handle = "kaizi"
                            password = "123456"
                            mode = .login
                            message = ""
                        } label: {
                            HStack {
                                Image(systemName: "bolt.fill")
                                Text(L("demoAccount", appLanguage))
                            }
                            .font(.subheadline.weight(.semibold))
                            .foregroundStyle(.primary)
                            .frame(maxWidth: .infinity)
                            .frame(height: 46)
                            .background(Color(.secondarySystemBackground))
                            .clipShape(RoundedRectangle(cornerRadius: 14))
                        }

                        AuthFeatureStrip()
                    }
                    .padding(18)
                    .background(Color(.systemBackground))
                    .clipShape(RoundedRectangle(cornerRadius: 22))
                    .shadow(color: .black.opacity(0.08), radius: 20, y: 10)
                }
                .padding()
            }
        }
        .animation(.snappy, value: mode)
    }

    private func submit() {
        message = ""

        guard !normalizedHandle.isEmpty, !password.isEmpty else {
            message = L("enterUserPass", appLanguage)
            return
        }

        if mode == .register && password.count < 6 {
            message = L("passwordTooShort", appLanguage)
            return
        }

        switch mode {
        case .login:
            let enteredPassword = password.trimmingCharacters(in: .whitespacesAndNewlines)
            guard let user = users.first(where: { $0.handle == normalizedHandle && $0.password == enteredPassword }) else {
                message = L("wrongCredentials", appLanguage)
                return
            }
            currentUserID = user.id

        case .register:
            let enteredPassword = password.trimmingCharacters(in: .whitespacesAndNewlines)
            guard !displayName.trimmingCharacters(in: .whitespacesAndNewlines).isEmpty else {
                message = L("enterDisplayName", appLanguage)
                return
            }
            guard users.allSatisfy({ $0.handle != normalizedHandle }) else {
                message = L("handleTaken", appLanguage)
                return
            }

            let user = AppUser(
                handle: normalizedHandle,
                displayName: displayName.trimmingCharacters(in: .whitespacesAndNewlines),
                password: enteredPassword,
                bio: L("noBio", appLanguage),
                avatarSymbol: "person.fill",
                avatarColorName: "indigo"
            )
            modelContext.insert(user)
            do {
                try modelContext.save()
                currentUserID = user.id
            } catch {
                modelContext.delete(user)
                message = L("databaseSaveFailed", appLanguage)
            }
        }
    }
}

        case .register: L("register", language)
        }
    }
}

private struct PasswordStrengthView: View {
    @Environment(\.appLanguage) private var appLanguage
    let password: String

    private var score: Int {
        var value = 0
        if password.count >= 6 { value += 1 }
        if password.rangeOfCharacter(from: .decimalDigits) != nil { value += 1 }
        if password.rangeOfCharacter(from: .uppercaseLetters) != nil { value += 1 }
        return value
    }

    var body: some View {
        VStack(alignment: .leading, spacing: 8) {
            HStack(spacing: 6) {
                ForEach(0..<3, id: \.self) { index in
                    Capsule()
                        .fill(index < score ? strengthColor : Color(.systemGray5))
                        .frame(height: 5)
                }
            }

            Text(strengthText)
                .font(.caption)
                .foregroundStyle(.secondary)
        }
    }

    private var strengthText: String {
        switch score {
        case 0: L("passwordWeak0", appLanguage)
        case 1: L("passwordWeak1", appLanguage)
        case 2: L("passwordWeak2", appLanguage)
        default: L("passwordWeak3", appLanguage)
        }
    }

    private var strengthColor: Color {
        switch score {
        case 0, 1: .orange
        case 2: .blue
        default: .green
        }
    }
}

private struct HomeTimelineView: View {
    @Environment(\.modelContext) private var modelContext
    @Environment(\.appLanguage) private var appLanguage
    let currentUser: AppUser
    @Binding var selectedTab: AppTab
    @Binding var isShowingComposer: Bool
    @Query(sort: \FeedPost.createdAt, order: .reverse) private var posts: [FeedPost]
    @Query(sort: \AppUser.joinedAt) private var users: [AppUser]
    @Query private var interactions: [PostInteraction]
    @Query private var comments: [PostComment]
    @Query private var follows: [FollowRelation]
    @State private var timelineMode = TimelineMode.recommend
    @State private var isShowingDrawer = false

    private var orderedPosts: [FeedPost] {
        let scoped: [FeedPost]
        switch timelineMode {
        case .recommend:
            scoped = posts
        case .following:
            let followingIDs = Set(follows.filter { $0.followerID == currentUser.id }.map(\.followingID))
            scoped = posts.filter { followingIDs.contains($0.authorID) || $0.authorID == currentUser.id }
        case .hot:
            scoped = posts
        }
        return scoped.sorted {
            timelineMode == .hot ? heat(for: $0) > heat(for: $1) : $0.createdAt > $1.createdAt
        }
    }

    var body: some View {
        NavigationStack {
            ZStack(alignment: .bottomTrailing) {
                VStack(spacing: 0) {
                    HomeHeaderView(
                        currentUser: currentUser,
                        selection: $timelineMode,
                        isShowingDrawer: $isShowingDrawer,
                        selectedTab: $selectedTab
                    )

                    List {
                        ForEach(orderedPosts) { post in
                            PostRow(
                                post: post,
                                currentUser: currentUser,
                                users: users,
                                interactions: interactions,
                                comments: comments,
                                heat: heat(for: post)
                            )
                            .listRowInsets(EdgeInsets())
                            .listRowSeparator(.hidden)
                            .listRowBackground(Color.clear)
                        }
                    }
                    .listStyle(.plain)
                    .scrollContentBackground(.hidden)
                    .background(KaiXTheme.pageBackground)
                    .safeAreaPadding(.top, 8)
                    .safeAreaPadding(.bottom, KaiXTheme.bottomContentPadding + 156)
                }
                .background(KaiXTheme.pageBackground)

                Button {
                    isShowingComposer = true
                } label: {
                    Image(systemName: "plus")
                        .font(.system(size: 22, weight: .semibold))
                        .foregroundStyle(.white)
                        .frame(width: 56, height: 56)
                        .background(.black)
                        .clipShape(Circle())
                        .shadow(color: .black.opacity(0.16), radius: 9, y: 5)
                }
                .accessibilityLabel(L("publishPost", appLanguage))
                .buttonStyle(PressScaleButtonStyle())
                .padding(.trailing, 24)
                .padding(.bottom, 78)
            }
            .toolbar(.hidden, for: .navigationBar)
            .accountDrawer(currentUser: currentUser, isPresented: $isShowingDrawer)
        }
    }

    private func heat(for post: FeedPost) -> Int {
        let likeCount = interactions.count { $0.postID == post.id && $0.kind == InteractionKind.like.rawValue }
        let bookmarkCount = interactions.count { $0.postID == post.id && $0.kind == InteractionKind.bookmark.rawValue }
        let ageHours = max(1, Int(Date.now.timeIntervalSince(post.createdAt) / 3600))
        return max(0, post.viewCount + likeCount * 120 + bookmarkCount * 40 - ageHours * 12)
    }

}

    }

private struct PostRow: View {
    @Environment(\.modelContext) private var modelContext
    @Environment(\.appLanguage) private var appLanguage
    let post: FeedPost
    let currentUser: AppUser
    let users: [AppUser]
    let interactions: [PostInteraction]
    let comments: [PostComment]
    let heat: Int
    @State private var isShowingDetail = false
    @State private var isShowingMediaPreview = false
    @State private var selectedAuthor: AppUser?

    private var author: AppUser? {
        users.first { $0.id == post.authorID }
    }

    private var likeCount: Int {
        interactions.count { $0.postID == post.id && $0.kind == InteractionKind.like.rawValue }
    }

    private var repostCount: Int {
        interactions.count { $0.postID == post.id && $0.kind == InteractionKind.repost.rawValue }
    }

    private var commentCount: Int {
        comments.count { $0.postID == post.id }
    }

    private var isLiked: Bool {
        interactions.contains { $0.postID == post.id && $0.userID == currentUser.id && $0.kind == InteractionKind.like.rawValue }
    }

    private var isReposted: Bool {
        interactions.contains { $0.postID == post.id && $0.userID == currentUser.id && $0.kind == InteractionKind.repost.rawValue }
    }

    private var isBookmarked: Bool {
        interactions.contains { $0.postID == post.id && $0.userID == currentUser.id && $0.kind == InteractionKind.bookmark.rawValue }
    }

    var body: some View {
        rowContent
        .listRowBackground(Color(.systemGroupedBackground))
        .fullScreenCover(isPresented: $isShowingDetail) {
            PostDetailView(post: post, currentUser: currentUser)
        }
        .fullScreenCover(isPresented: $isShowingMediaPreview) {
            MediaPreviewView(post: post)
        }
        .fullScreenCover(item: $selectedAuthor) { user in
            UserProfileView(currentUser: currentUser, user: user)
        }
    }

    private var rowContent: some View {
        VStack(alignment: .leading, spacing: post.mediaKind == .none ? 15 : 17) {
            authorHeader

            Button {
                openDetail()
            } label: {
                Text(post.bodyWithoutTrailingTags)
                    .font(.body)
                    .lineSpacing(4)
                    .foregroundStyle(.primary)
                    .frame(maxWidth: .infinity, alignment: .leading)
                    .fixedSize(horizontal: false, vertical: true)
            }
            .buttonStyle(.plain)

            if !post.tags.isEmpty {
                TagWrapView(tags: post.tags)
            }

                Button {
                    isShowingMediaPreview = true
                } label: {
                    PostMediaPreview(post: post)
                }
                .buttonStyle(.plain)
            }

            HStack(spacing: 0) {
                ActionButton(icon: "bubble.left", value: "\(commentCount)", isActive: false) {
                    openDetail()
                }
                Spacer()
                ActionButton(icon: "arrow.2.squarepath", value: "\(repostCount)", isActive: isReposted, activeColor: .green) {
                    toggle(.repost)
                }
                Spacer()
                ActionButton(icon: isLiked ? "heart.fill" : "heart", value: "\(likeCount)", isActive: isLiked, activeColor: .pink) {
                    toggle(.like)
                }
                Spacer()
                ActionButton(icon: isBookmarked ? "bookmark.fill" : "bookmark", value: "\(bookmarkCount)", isActive: isBookmarked, activeColor: .blue) {
                    toggle(.bookmark)
                }
                Spacer()
                HeatMetricView(heat: heat)
            }
            .padding(.top, post.mediaKind == .none ? 2 : 4)
            .padding(.bottom, 2)
        }
        .padding(.horizontal, 18)
        .padding(.top, 18)
        .padding(.bottom, 16)
        .background(Color(.systemBackground))
        .clipShape(RoundedRectangle(cornerRadius: 22))
        .overlay(
            RoundedRectangle(cornerRadius: 22)
                .stroke(Color.black.opacity(0.04), lineWidth: 0.8)

        .padding(.horizontal, 16)
        .padding(.vertical, post.mediaKind == .none ? 8 : 10)

                .buttonStyle(.plain)

                            selectedAuthor = author

                            HStack(spacing: 5) {
                                Text(author.displayName)
                                    .font(.subheadline.weight(.bold))
                                    .lineLimit(1)

                                        .font(.caption)
                                        .foregroundStyle(.blue)
                                }
                            }
                        }
                        .buttonStyle(.plain)
                    } else {
                        Text(L("unknownUser", appLanguage))
                            .font(.subheadline.weight(.bold))
                    }
                }

                if let author {
                    Button {
                        selectedAuthor = author
                    } label: {
                        Text("@\(author.handle) · \(post.createdAt.relativeText)")
                            .font(.caption)
                            .foregroundStyle(.secondary)
                            .lineLimit(1)
                    }
                    .buttonStyle(.plain)
                } else {
                    Text("@unknown · \(post.createdAt.relativeText)")
                        .font(.caption)
                        .foregroundStyle(.secondary)
                }

                if let repostOfID = post.repostOfID {
                    Label("\(L("reposted", appLanguage)) \(shortID(repostOfID))", systemImage: "arrow.2.squarepath")
                        .font(.caption)
                        .foregroundStyle(.green)
                }
            }
        }
    }

    private func openDetail() {
        post.viewCount += 1
        try? modelContext.save()
        isShowingDetail = true
    }

    private var bookmarkCount: Int {
        interactions.count { $0.postID == post.id && $0.kind == InteractionKind.bookmark.rawValue }
    }

    private func toggle(_ kind: InteractionKind) {
        if let existing = interactions.first(where: { $0.postID == post.id && $0.userID == currentUser.id && $0.kind == kind.rawValue }) {
            modelContext.delete(existing)
        } else {
            modelContext.insert(PostInteraction(postID: post.id, userID: currentUser.id, kind: kind))
            if kind == .repost {
                modelContext.insert(FeedPost(authorID: currentUser.id, content: post.content, repostOfID: post.id, viewCount: 0))
                Task {
                    await SharedBackend.createPost(userID: currentUser.id, content: post.content, repostOfID: post.id)
                }
            }
        }

        try? modelContext.save()
        Task {
            await SharedBackend.togglePostInteraction(postID: post.id, userID: currentUser.id, kind: kind)
        }
    }

    private func shortID(_ id: String) -> String {
        "#\(id.prefix(6))"
    }
}

private struct PostMediaPreview: View {
    let post: FeedPost

    var body: some View {
        ZStack(alignment: .bottomTrailing) {
            if !post.imageDataItems.isEmpty {
                let images = post.imageDataItems
                LazyVGrid(columns: Array(repeating: GridItem(.flexible(), spacing: 6), count: images.count == 1 ? 1 : 2), spacing: 6) {
                    ForEach(images, id: \.self) { data in
                        if let image = UIImage(data: data) {
                            Image(uiImage: image)
                                .resizable()
                                .scaledToFill()
                                .frame(height: images.count == 1 ? 188 : 106)
                                .frame(maxWidth: .infinity)
                                .clipped()
                        }
                    }
                }
            } else if post.videoData != nil {
                mediaTile(symbol: "play.rectangle.fill", opacity: 0.9)
                    .aspectRatio(16 / 9, contentMode: .fit)
            } else {
                if post.mediaKind == .gallery {
                    LazyVGrid(columns: Array(repeating: GridItem(.flexible(), spacing: 6), count: 2), spacing: 6) {
                        ForEach(0..<4, id: \.self) { index in
                            mediaTile(symbol: index == 0 ? post.mediaSymbol : ["photo.fill", "sparkles", "rectangle.stack.fill"][index - 1], opacity: 0.86 - Double(index) * 0.08)
                                .aspectRatio(1.28, contentMode: .fit)
                        }
                    }
                } else {
                    mediaTile(symbol: post.mediaSymbol.isEmpty ? "photo.fill" : post.mediaSymbol, opacity: 0.9)
                        .aspectRatio(post.mediaKind == .video ? 16 / 9 : 1.7, contentMode: .fit)
                }
            }

            if post.mediaKind == .video {
                HStack(spacing: 6) {
                    Image(systemName: "play.fill")
                    Text(post.videoDuration.isEmpty ? "00:30" : post.videoDuration)
                }
                .font(.caption2.weight(.bold))
                .foregroundStyle(.white)
                .padding(.horizontal, 9)
                .padding(.vertical, 6)
                .background(.black.opacity(0.72))
                .clipShape(Capsule())
                .padding(10)
            }
        }
        .clipShape(RoundedRectangle(cornerRadius: 18))
        .frame(maxHeight: post.mediaKind == .gallery ? 214 : 190)
        .clipped()
        .overlay(
            RoundedRectangle(cornerRadius: 18)
                .stroke(Color.black.opacity(0.04), lineWidth: 0.8)
        )
    }

    private func mediaTile(symbol: String, opacity: Double) -> some View {
        ZStack {
            LinearGradient(
                colors: [Color.named(post.mediaColorName).opacity(opacity), Color(.darkGray).opacity(0.82)],
                startPoint: .topLeading,
                endPoint: .bottomTrailing
            )
            Image(systemName: symbol)
                .font(.system(size: 34, weight: .bold))
                .foregroundStyle(.white.opacity(0.92))
            if !post.mediaTitle.isEmpty {
                VStack {
                    Spacer()
                    Text(post.mediaTitle)
                        .font(.caption.weight(.bold))
                        .foregroundStyle(.white)
                        .lineLimit(1)
                        .padding(.horizontal, 10)
                        .padding(.vertical, 8)
                        .frame(maxWidth: .infinity, alignment: .leading)
                        .background(.black.opacity(0.22))
                }
            }
        }
    }
}

private struct PostDetailView: View {
    @Environment(\.modelContext) private var modelContext
    @Environment(\.dismiss) private var dismiss
    @Environment(\.appLanguage) private var appLanguage
    let post: FeedPost
    let currentUser: AppUser
    @Query(sort: \AppUser.joinedAt) private var users: [AppUser]
    @Query(sort: \PostComment.createdAt) private var comments: [PostComment]
    @Query private var interactions: [PostInteraction]
    @State private var commentText = ""
    @State private var replyingTo: PostComment?

    private var postComments: [PostComment] {
        comments.filter { $0.postID == post.id }
    }

    var body: some View {
        NavigationStack {
            VStack(spacing: 0) {
                StickyNavigationBar(title: L("postDetails", appLanguage), onBack: { dismiss() })

                List {
                    PostSummaryCard(post: post, currentUser: currentUser, users: users, interactions: interactions, comments: comments)
                        .listRowInsets(EdgeInsets(top: 12, leading: 14, bottom: 8, trailing: 14))
                        .listRowSeparator(.hidden)
                        .listRowBackground(Color.clear)

                    Section {
                        HStack {
                            Text(L("comments", appLanguage))
                                .font(.headline.weight(.black))
                            Spacer()
                            Text(L("newest", appLanguage))
                                .font(.caption.weight(.bold))
                                .foregroundStyle(.secondary)
                                .padding(.horizontal, 10)
                                .padding(.vertical, 6)
                                .background(Color(.secondarySystemBackground))
                                .clipShape(Capsule())
                        }
                        .listRowInsets(EdgeInsets(top: 10, leading: 16, bottom: 4, trailing: 16))
                        .listRowSeparator(.hidden)
                        .listRowBackground(Color.clear)

                        if postComments.isEmpty {
                            ContentUnavailableView(L("noComments", appLanguage), systemImage: "bubble.left", description: Text(L("firstComment", appLanguage)))
                                .listRowBackground(Color.clear)
                        } else {
                            ForEach(postComments) { comment in
                                CommentRow(
                                    comment: comment,
                                    currentUser: currentUser,
                                    users: users,
                                    interactions: interactions,
                                    onReply: { replyingTo = comment }
                                )
                                .listRowInsets(EdgeInsets(top: 4, leading: 14, bottom: 4, trailing: 14))
                                .listRowSeparator(.hidden)
                                .listRowBackground(Color.clear)
                            }
                        }
                    }
                }
                .scrollContentBackground(.hidden)
                .listStyle(.plain)
                .background(KaiXTheme.pageBackground)
                .safeAreaPadding(.bottom, 76)
            }
            .background(KaiXTheme.pageBackground)
            .safeAreaInset(edge: .bottom) {
                CommentInputBar(
                    currentUser: currentUser,
                    text: $commentText,
                    replyingToHandle: replyingTo.map { userHandle($0.authorID) },
                    onCancelReply: { replyingTo = nil },
                    onSend: sendComment
                )
            }
            .toolbar(.hidden, for: .navigationBar)
        }
    }

    private func sendComment() {
        let trimmed = commentText.trimmingCharacters(in: .whitespacesAndNewlines)
        guard !trimmed.isEmpty else { return }
        modelContext.insert(PostComment(postID: post.id, authorID: currentUser.id, content: trimmed, replyToUserID: replyingTo?.authorID))
        let replyToUserID = replyingTo?.authorID
        commentText = ""
        replyingTo = nil
        try? modelContext.save()
        Task {
            await SharedBackend.createComment(postID: post.id, userID: currentUser.id, content: trimmed, replyToUserID: replyToUserID)
        }
    }

    private func userHandle(_ id: String) -> String {
        users.first { $0.id == id }?.handle ?? "unknown"
    }
}

private struct PostSummaryCard: View {
    @Environment(\.modelContext) private var modelContext
    @Environment(\.appLanguage) private var appLanguage
    let post: FeedPost
    let currentUser: AppUser
    let users: [AppUser]
    let interactions: [PostInteraction]
    let comments: [PostComment]
    @State private var isShowingMediaPreview = false
    @State private var selectedAuthor: AppUser?

    private var author: AppUser? {
        users.first { $0.id == post.authorID }
    }

    private var likeCount: Int { interactions.count { $0.postID == post.id && $0.kind == InteractionKind.like.rawValue } }
    private var repostCount: Int { interactions.count { $0.postID == post.id && $0.kind == InteractionKind.repost.rawValue } }
    private var bookmarkCount: Int { interactions.count { $0.postID == post.id && $0.kind == InteractionKind.bookmark.rawValue } }
    private var commentCount: Int { comments.count { $0.postID == post.id } }
    private var isLiked: Bool { interactions.contains { $0.postID == post.id && $0.userID == currentUser.id && $0.kind == InteractionKind.like.rawValue } }
    private var isReposted: Bool { interactions.contains { $0.postID == post.id && $0.userID == currentUser.id && $0.kind == InteractionKind.repost.rawValue } }
    private var isBookmarked: Bool { interactions.contains { $0.postID == post.id && $0.userID == currentUser.id && $0.kind == InteractionKind.bookmark.rawValue } }
    private var heatScore: Int {
        let ageHours = max(1, Int(Date.now.timeIntervalSince(post.createdAt) / 3600))
        return max(0, post.viewCount + commentCount * 90 + repostCount * 140 + likeCount * 120 + bookmarkCount * 40 - ageHours * 12)
    }

    var body: some View {
        VStack(alignment: .leading, spacing: 14) {
            HStack(spacing: 10) {
                if let author {
                    Button {
                        selectedAuthor = author
                    } label: {
                        AvatarView(user: author, size: 46)
                    }
                    .buttonStyle(.plain)
                } else {
                    AvatarPlaceholder(size: 46)
                }
                VStack(alignment: .leading, spacing: 2) {
                    HStack(spacing: 4) {
                        if let author {
                            Button {
                                selectedAuthor = author
                            } label: {
                                HStack(spacing: 4) {
                                    Text(author.displayName)
                                        .font(.headline.weight(.bold))
                                    if author.isVerified {
                                        Image(systemName: "checkmark.seal.fill")
                                            .font(.caption)
                                            .foregroundStyle(.blue)
                                    }
                                }
                            }
                            .buttonStyle(.plain)
                        } else {
                            Text(L("unknownUser", appLanguage))
                                .font(.headline.weight(.bold))
                        }
                    }
                    Text("@\(author?.handle ?? "unknown") · \(post.createdAt.relativeText)")
                        .font(.caption)
                        .foregroundStyle(.secondary)
                }
                Spacer()
            }

            Text(post.bodyWithoutTrailingTags)
                .font(.system(size: 18, weight: .regular))
                .lineSpacing(5)
                .fixedSize(horizontal: false, vertical: true)

            if !post.tags.isEmpty {
                TagWrapView(tags: post.tags)
            }

            if post.mediaKind != .none {
                Button {
                    isShowingMediaPreview = true
                } label: {
                    PostMediaPreview(post: post)
                }
                .buttonStyle(.plain)
            }

            VStack(alignment: .leading, spacing: 8) {
                Text(post.createdAt.formatted(date: .numeric, time: .shortened))
                    .font(.caption)
                    .foregroundStyle(.secondary)
                Text("\(post.viewCount.formatted()) \(L("views", appLanguage))")
                    .font(.caption)
                    .foregroundStyle(.secondary)
                HStack(spacing: 0) {
                    detailStat(value: commentCount, title: L("comments", appLanguage))
                    detailStat(value: repostCount, title: L("reposted", appLanguage))
                    detailStat(value: likeCount, title: L("likes", appLanguage))
                    detailStat(value: bookmarkCount, title: L("bookmarks", appLanguage))
                    detailStat(value: heatScore, title: L("heat", appLanguage))
                }
                .padding(10)
                .background(Color(.secondarySystemBackground))
                .clipShape(RoundedRectangle(cornerRadius: 14))
            }
            .padding(.vertical, 8)

            HStack {
                ActionButton(icon: "bubble.left", value: "", isActive: false) {}
                Spacer()
                ActionButton(icon: "arrow.2.squarepath", value: "", isActive: isReposted, activeColor: .green) { toggle(.repost) }
                Spacer()
                ActionButton(icon: isLiked ? "heart.fill" : "heart", value: "", isActive: isLiked, activeColor: .pink) { toggle(.like) }
                Spacer()
                ActionButton(icon: isBookmarked ? "bookmark.fill" : "bookmark", value: "", isActive: isBookmarked, activeColor: .blue) { toggle(.bookmark) }
                Spacer()
                ActionButton(icon: "square.and.arrow.up", value: "", isActive: false) {}
            }
            .padding(.top, 2)
            .overlay(alignment: .top) {
                Divider().offset(y: -8)
            }
        }
        .padding(16)
        .background(KaiXTheme.cardBackground)
        .clipShape(RoundedRectangle(cornerRadius: 22))
        .overlay(
            RoundedRectangle(cornerRadius: 22)
                .stroke(KaiXTheme.line, lineWidth: 0.8)
        )
        .fullScreenCover(isPresented: $isShowingMediaPreview) {
            MediaPreviewView(post: post)
        }
        .fullScreenCover(item: $selectedAuthor) { user in
            UserProfileView(currentUser: currentUser, user: user)
        }
    }

    private func detailStat(value: Int, title: String) -> some View {
        VStack(alignment: .leading, spacing: 2) {
            Text(value.formatted())
                .font(.subheadline.weight(.black))
            Text(title)
                .font(.caption2)
                .foregroundStyle(.secondary)
        }
        .frame(maxWidth: .infinity, alignment: .leading)
    }

    private func toggle(_ kind: InteractionKind) {
        if let existing = interactions.first(where: { $0.postID == post.id && $0.userID == currentUser.id && $0.kind == kind.rawValue }) {
            modelContext.delete(existing)
        } else {
            modelContext.insert(PostInteraction(postID: post.id, userID: currentUser.id, kind: kind))
            if kind == .repost {
                modelContext.insert(FeedPost(authorID: currentUser.id, content: post.content, repostOfID: post.id, viewCount: 0))
            }
        }
        try? modelContext.save()
    }
}

private struct CommentRow: View {
    @Environment(\.modelContext) private var modelContext
    @Environment(\.appLanguage) private var appLanguage
    let comment: PostComment
    let currentUser: AppUser
    let users: [AppUser]
    let interactions: [PostInteraction]
    let onReply: () -> Void
    @State private var selectedAuthor: AppUser?

    private var author: AppUser? {
        users.first { $0.id == comment.authorID }
    }

    private var replyTarget: AppUser? {
        guard let replyToUserID = comment.replyToUserID else { return nil }
        return users.first { $0.id == replyToUserID }
    }

    private var likeCount: Int {
        interactions.count { $0.postID == comment.id && $0.kind == InteractionKind.commentLike.rawValue }
    }

    private var isLiked: Bool {
        interactions.contains { $0.postID == comment.id && $0.userID == currentUser.id && $0.kind == InteractionKind.commentLike.rawValue }
    }

    var body: some View {
        HStack(alignment: .top, spacing: 12) {
            if let author {
                Button {
                    selectedAuthor = author
                } label: {
                    AvatarView(user: author, size: 38)
                }
                .buttonStyle(.plain)
            } else {
                AvatarPlaceholder(size: 38)
            }

            VStack(alignment: .leading, spacing: 4) {
                HStack(spacing: 5) {
                    if let author {
                        Button {
                            selectedAuthor = author
                        } label: {
                            HStack(spacing: 5) {
                                Text(author.displayName)
                                    .font(.subheadline.weight(.semibold))
                                if author.isVerified {
                                    Image(systemName: "checkmark.seal.fill")
                                        .font(.caption)
                                        .foregroundStyle(.blue)
                                }
                                Text("@\(author.handle)")
                                    .font(.caption)
                                    .foregroundStyle(.secondary)
                            }
                        }
                        .buttonStyle(.plain)
                    } else {
                        Text(L("unknownUser", appLanguage))
                            .font(.subheadline.weight(.semibold))
                    }
                }

                if let replyTarget {
                    Text("\(L("replyTo", appLanguage)) @\(replyTarget.handle)")
                        .font(.caption)
                        .foregroundStyle(.blue)
                }

                Text(comment.content)

                HStack(spacing: 18) {
                    Text(comment.createdAt.relativeText)
                        .font(.caption)
                        .foregroundStyle(.secondary)

                    Button(L("replyTo", appLanguage)) {
                        onReply()
                    }
                    .font(.caption.weight(.semibold))
                    .buttonStyle(.plain)

                    Button {
                        toggleLike()
                    } label: {
                        Label("\(likeCount)", systemImage: isLiked ? "heart.fill" : "heart")
                            .font(.caption.weight(.semibold))
                            .foregroundStyle(isLiked ? .pink : .secondary)
                    }
                    .buttonStyle(.plain)
                }
            }
        }
        .padding(12)
        .background(KaiXTheme.cardBackground)
        .clipShape(RoundedRectangle(cornerRadius: 14))
        .overlay(
            RoundedRectangle(cornerRadius: 14)
                .stroke(KaiXTheme.line, lineWidth: 0.8)
        )
        .padding(.vertical, 3)
        .fullScreenCover(item: $selectedAuthor) { user in
            UserProfileView(currentUser: currentUser, user: user)
        }
    }

    private func toggleLike() {
        if let existing = interactions.first(where: { $0.postID == comment.id && $0.userID == currentUser.id && $0.kind == InteractionKind.commentLike.rawValue }) {
            modelContext.delete(existing)
        } else {
            modelContext.insert(PostInteraction(postID: comment.id, userID: currentUser.id, kind: .commentLike))
        }
        try? modelContext.save()
        Task {
            await SharedBackend.toggleCommentLike(commentID: comment.id, userID: currentUser.id)
        }
    }
}

private struct ExploreView: View {
    @Environment(\.appLanguage) private var appLanguage
    let currentUser: AppUser
    @Query(sort: \FeedPost.createdAt, order: .reverse) private var posts: [FeedPost]
    @Query(sort: \AppUser.joinedAt) private var users: [AppUser]
    @Query private var interactions: [PostInteraction]
    @Query private var comments: [PostComment]
    @Query private var follows: [FollowRelation]
    @State private var searchText = ""
    @State private var scope = SearchScope.hot
    @State private var isShowingDrawer = false
    @State private var selectedDetail: SearchDetailSelection?

    private var filteredPosts: [FeedPost] {
        let query = searchText.trimmingCharacters(in: .whitespacesAndNewlines)
        guard !query.isEmpty else { return posts.sorted { heat(for: $0) > heat(for: $1) } }

        return posts.filter { post in
            post.content.localizedCaseInsensitiveContains(query)
            || author(for: post)?.displayName.localizedCaseInsensitiveContains(query) == true
            || author(for: post)?.handle.localizedCaseInsensitiveContains(query) == true
        }
    }

    private var filteredUsers: [AppUser] {
        let query = searchText.trimmingCharacters(in: .whitespacesAndNewlines)
        guard !query.isEmpty else { return users.filter { $0.id != currentUser.id } }

        return users.filter {
            $0.displayName.localizedCaseInsensitiveContains(query)
            || $0.handle.localizedCaseInsensitiveContains(query)
            || $0.bio.localizedCaseInsensitiveContains(query)
        }
    }

    private var hotRanking: [RankingItem] {
        posts
            .sorted { heat(for: $0) > heat(for: $1) }
            .prefix(10)
            .enumerated()
            .map { offset, post in
                RankingItem(
                    rank: offset + 1,
                    title: post.content.headlineText,
                    subtitle: "\(author(for: post)?.displayName ?? L("unknownUser", appLanguage)) · \(post.createdAt.relativeText)",
                    badge: post.isRising ? L("newBadge", appLanguage) : nil,
                    heat: heat(for: post),
                    kind: .post(post)
                )
            }
    }

    private var newsRanking: [RankingItem] {
        posts
            .filter { $0.content.contains("新闻") || $0.content.contains("发布") || $0.content.contains("上线") || $0.content.contains("开放") }
            .sorted { heat(for: $0) > heat(for: $1) }
            .prefix(8)
            .enumerated()
            .map { offset, post in
                RankingItem(
                    rank: offset + 1,
                    title: post.content.headlineText,
                    subtitle: "\(post.viewCount.formatted()) \(L("views", appLanguage)) · \(post.createdAt.relativeText)",
                    badge: offset < 2 ? L("hotBadge", appLanguage) : nil,
                    heat: heat(for: post),
                    kind: .post(post)
                )
            }
    }

    private var topicRanking: [RankingItem] {
        let tags = posts.flatMap { post in
            post.content
                .split(separator: " ")
                .map(String.init)
                .filter { $0.hasPrefix("#") && $0.count > 1 }
        }

        let grouped = Dictionary(grouping: tags, by: { $0.lowercased() })
        return grouped.map { key, values -> TopicStats in
            let related = posts.filter { $0.content.lowercased().contains(key) }
            let likeCount = related.reduce(0) { total, post in
                total + interactions.count { $0.postID == post.id && $0.kind == InteractionKind.like.rawValue }
            }
            let viewCount = related.reduce(0) { $0 + $1.viewCount }
            return TopicStats(title: key, postCount: values.count, heat: viewCount + likeCount * 160 + values.count * 80)
        }
        .sorted { $0.heat > $1.heat }
        .prefix(10)
        .enumerated()
        .map { offset, topic in
            RankingItem(
                rank: offset + 1,
                title: topic.title,
                subtitle: "\(topic.postCount) \(L("relatedPostCount", appLanguage))",
                badge: offset < 3 ? L("hotBadge", appLanguage) : nil,
                heat: topic.heat,
                kind: .topic(topic.title)
            )
        }
    }

    private var topTopics: [RankingItem] {
        Array(topicRanking.prefix(8))
    }

    var body: some View {
        NavigationStack {
            List {
                Section {
                    SearchTopBar(
                        title: L("search", appLanguage),
                        currentUser: currentUser,
                        onBack: nil,
                        isShowingDrawer: $isShowingDrawer
                    )
                    .listRowInsets(EdgeInsets(top: 8, leading: 18, bottom: 8, trailing: 18))
                    .listRowSeparator(.hidden)
                    .listRowBackground(Color.clear)

                    SearchHeaderField(text: $searchText, prompt: L("searchPrompt", appLanguage))
                        .listRowInsets(EdgeInsets(top: 0, leading: 18, bottom: 10, trailing: 18))
                        .listRowSeparator(.hidden)
                        .listRowBackground(Color.clear)
                }

                if searchText.trimmingCharacters(in: .whitespacesAndNewlines).isEmpty {
                    Section {
                        Button {
                            selectedDetail = SearchDetailSelection(title: L("forYouHeat", appLanguage), posts: Array(filteredPosts.prefix(12)))
                        } label: {
                            HappeningNowCard(totalPosts: posts.count, totalUsers: users.count, hottest: hotRanking.first)
                        }
                        .buttonStyle(.plain)
                        .listRowInsets(EdgeInsets(top: 2, leading: 18, bottom: 8, trailing: 18))
                        .listRowSeparator(.hidden)
                        .listRowBackground(Color.clear)
                    }

                    Section {
                        TrendingTopicStrip(items: topTopics) { topic in
                            searchText = topic
                        }
                        .listRowInsets(EdgeInsets(top: 0, leading: 18, bottom: 6, trailing: 0))
                        .listRowSeparator(.hidden)
                        .listRowBackground(Color.clear)
                    }

                    Section {
                        RankingSegmentedControl(selection: $scope)
                            .listRowInsets(EdgeInsets(top: 0, leading: 18, bottom: 6, trailing: 18))
                            .listRowSeparator(.hidden)
                            .listRowBackground(Color.clear)
                    }

                    Section {
                        SearchSectionHeader(title: scope.sectionTitle(appLanguage), icon: scope.sectionIcon, tint: scope.tint)
                            .listRowInsets(EdgeInsets(top: 4, leading: 20, bottom: 4, trailing: 20))
                            .listRowSeparator(.hidden)
                            .listRowBackground(Color.clear)

                        if scope == .users {
                            ForEach(filteredUsers.prefix(10)) { user in
                                NavigationLink {
                                    UserProfileView(currentUser: currentUser, user: user)
                                } label: {
                                    UserResultRow(user: user, currentUser: currentUser, follows: follows)
                                }
                                .listRowBackground(Color.clear)
                            }
                        } else {
                            let ranking = rankingItems(for: scope)
                            if ranking.isEmpty {
                                ContentUnavailableView(L("noContent", appLanguage), systemImage: "chart.line.uptrend.xyaxis")
                            } else {
                                ForEach(ranking) { item in
                                    Button {
                                        selectedDetail = SearchDetailSelection(title: item.title, posts: detailPosts(for: item))
                                    } label: {
                                        RankingRow(item: item)
                                    }
                                    .buttonStyle(.plain)
                                    .listRowInsets(EdgeInsets(top: 5, leading: 18, bottom: 5, trailing: 18))
                                    .listRowSeparator(.hidden)
                                    .listRowBackground(Color.clear)
                                }
                            }
                        }
                    }

                    Section(L("recommendedFollow", appLanguage)) {
                        ForEach(filteredUsers.prefix(4)) { user in
                            NavigationLink {
                                UserProfileView(currentUser: currentUser, user: user)
                            } label: {
                                UserResultRow(user: user, currentUser: currentUser, follows: follows)
                            }
                            .listRowBackground(Color.clear)
                        }
                    }
                } else {
                    Section(L("searchResults", appLanguage)) {
                        SearchSummary(query: searchText, postCount: filteredPosts.count, userCount: filteredUsers.count)
                            .listRowBackground(Color.clear)
                    }

                    Section(L("relatedPosts", appLanguage)) {
                        if filteredPosts.isEmpty {
                            ContentUnavailableView(L("noPostsFound", appLanguage), systemImage: "doc.text.magnifyingglass")
                        } else {
                            ForEach(filteredPosts) { post in
                        PostRow(post: post, currentUser: currentUser, users: users, interactions: interactions, comments: comments, heat: heat(for: post))
                            .listRowInsets(EdgeInsets())
                            .listRowBackground(Color.clear)
                    }
                }
                    }

                    Section(L("relatedUsers", appLanguage)) {
                        if filteredUsers.isEmpty {
                            ContentUnavailableView(L("noUsersFound", appLanguage), systemImage: "person.crop.circle.badge.questionmark")
                        } else {
                            ForEach(filteredUsers) { user in
                                NavigationLink {
                                    UserProfileView(currentUser: currentUser, user: user)
                                } label: {
                                    UserResultRow(user: user, currentUser: currentUser, follows: follows)
                                }
                                .listRowBackground(Color.clear)
                            }
                        }
                    }
                }
            }
            .listStyle(.plain)
            .scrollContentBackground(.hidden)
            .background(KaiXTheme.pageBackground)
            .safeAreaPadding(.bottom, KaiXTheme.bottomContentPadding)
            .toolbar(.hidden, for: .navigationBar)
            .accountDrawer(currentUser: currentUser, isPresented: $isShowingDrawer)
            .fullScreenCover(item: $selectedDetail) { detail in
                SearchDetailView(title: detail.title, posts: detail.posts, currentUser: currentUser)
            }
        }
    }

    private func rankingItems(for scope: SearchScope) -> [RankingItem] {
        switch scope {
        case .hot: hotRanking
        case .news: newsRanking.isEmpty ? hotRanking : newsRanking
        case .topics: topicRanking
        case .users: []
        }
    }

    private func author(for post: FeedPost) -> AppUser? {
        users.first { $0.id == post.authorID }
    }

    private func detailPosts(for item: RankingItem) -> [FeedPost] {
        switch item.kind {
        case .post(let post):
            return [post]
        case .topic(let topic):
            return posts
                .filter { $0.content.localizedCaseInsensitiveContains(topic) }
                .sorted { heat(for: $0) > heat(for: $1) }
        }
    }

    private func heat(for post: FeedPost) -> Int {
        let likeCount = interactions.count { $0.postID == post.id && $0.kind == InteractionKind.like.rawValue }
        let bookmarkCount = interactions.count { $0.postID == post.id && $0.kind == InteractionKind.bookmark.rawValue }
        let ageHours = max(1, Int(Date.now.timeIntervalSince(post.createdAt) / 3600))
        return max(0, post.viewCount + likeCount * 120 + bookmarkCount * 40 - ageHours * 12)
    }
}

private struct SearchDetailView: View {
    @Environment(\.dismiss) private var dismiss
    @Environment(\.appLanguage) private var appLanguage
    let title: String
    let posts: [FeedPost]
    let currentUser: AppUser
    @Query(sort: \AppUser.joinedAt) private var users: [AppUser]
    @Query private var interactions: [PostInteraction]
    @Query private var comments: [PostComment]
    @State private var drawerDisabled = false

    var body: some View {
        VStack(spacing: 0) {
            StickyNavigationBar(title: title, onBack: { dismiss() })

            List {
                Section {
                    HappeningNowCard(totalPosts: posts.count, totalUsers: relatedUserCount, hottest: topItem)
                        .listRowInsets(EdgeInsets(top: 12, leading: 16, bottom: 8, trailing: 16))
                        .listRowSeparator(.hidden)
                        .listRowBackground(Color.clear)
                }

                Section(L("relatedPosts", appLanguage)) {
                    if posts.isEmpty {
                        ContentUnavailableView(L("noPostsFound", appLanguage), systemImage: "doc.text.magnifyingglass")
                    } else {
                        ForEach(posts) { post in
                            PostRow(
                                post: post,
                                currentUser: currentUser,
                                users: users,
                                interactions: interactions,
                                comments: comments,
                                heat: heat(for: post)
                            )
                            .listRowInsets(EdgeInsets())
                            .listRowSeparator(.hidden)
                            .listRowBackground(Color.clear)
                        }
                    }
                }
            }
            .listStyle(.plain)
            .scrollContentBackground(.hidden)
            .background(KaiXTheme.pageBackground)
            .safeAreaPadding(.bottom, KaiXTheme.bottomContentPadding)
        }
        .background(KaiXTheme.pageBackground)
        .toolbar(.hidden, for: .navigationBar)
    }

    private var relatedUserCount: Int {
        Set(posts.map(\.authorID)).count
    }

    private var topItem: RankingItem? {
        guard let post = posts.sorted(by: { heat(for: $0) > heat(for: $1) }).first else { return nil }
        return RankingItem(
            rank: 1,
            title: post.content.headlineText,
            subtitle: "\(post.viewCount.formatted()) \(L("views", appLanguage)) · \(post.createdAt.relativeText)",
            badge: post.isRising ? L("newBadge", appLanguage) : nil,
            heat: heat(for: post),
            kind: .post(post)
        )
    }

    private func heat(for post: FeedPost) -> Int {
        let likeCount = interactions.count { $0.postID == post.id && $0.kind == InteractionKind.like.rawValue }
        let bookmarkCount = interactions.count { $0.postID == post.id && $0.kind == InteractionKind.bookmark.rawValue }
        let ageHours = max(1, Int(Date.now.timeIntervalSince(post.createdAt) / 3600))
        return max(0, post.viewCount + likeCount * 120 + bookmarkCount * 40 - ageHours * 12)
    }
}

private enum SearchScope: String, CaseIterable, Identifiable {
    case hot
    case news
    case topics
    case users

    var id: String { rawValue }

    func title(_ language: AppLanguage) -> String {
        switch self {
        case .hot: L("hotSearch", language)
        case .news: L("news", language)
        case .topics: L("topics", language)
        case .users:
            switch language {
            case .ja: "ユーザー"
            case .en: "Users"
            default: "用户"
            }
        }
    }

    func sectionTitle(_ language: AppLanguage) -> String {
        switch self {
        case .hot: L("hotRanking", language)
        case .news: L("newsRanking", language)
        case .topics: L("topicRanking", language)
        case .users:
            switch language {
            case .ja: "おすすめユーザー"
            case .en: "Recommended Users"
            default: "推荐用户"
            }
        }
    }

    var sectionIcon: String {
        switch self {
        case .hot: "flame.fill"
        case .news: "newspaper.fill"
        case .topics: "number"
        case .users: "person.2.fill"
        }
    }

    var tint: Color {
        switch self {
        case .hot: .orange
        case .news: .blue
        case .topics: .purple
        case .users: .green
        }
    }
}

private struct MessagesView: View {
    @Environment(\.modelContext) private var modelContext
    @Environment(\.appLanguage) private var appLanguage
    let currentUser: AppUser
    @Query(sort: \AppUser.joinedAt) private var users: [AppUser]
    @Query(sort: \DirectMessage.createdAt, order: .reverse) private var messages: [DirectMessage]
    @State private var selectedPeer: AppUser?
    @State private var isShowingDrawer = false

    private var peers: [AppUser] {
        let peerIDs = Set(messages.compactMap { message -> String? in
            if message.senderID == currentUser.id { return message.receiverID }
            if message.receiverID == currentUser.id { return message.senderID }
            return nil
        })

        let existing = users.filter { peerIDs.contains($0.id) && $0.id != currentUser.id }
        let available = users.filter { $0.id != currentUser.id && !peerIDs.contains($0.id) }
        return existing + available
    }

    var body: some View {
        NavigationStack {
            List {
                ForEach(peers) { peer in
                    Button {
                        selectedPeer = peer
                    } label: {
                        ConversationRow(peer: peer, currentUser: currentUser, messages: messages)
                    }
                    .buttonStyle(.plain)
                    .listRowInsets(EdgeInsets(top: 6, leading: 14, bottom: 6, trailing: 14))
                    .listRowSeparator(.hidden)
                    .listRowBackground(Color.clear)
                }
                .onDelete(perform: deleteConversation)
            }
            .listStyle(.plain)
            .scrollContentBackground(.hidden)
            .background(Color(.systemGroupedBackground))
            .safeAreaPadding(.bottom, KaiXTheme.bottomContentPadding)
            .navigationTitle(L("messages", appLanguage))
            .toolbar {
                ToolbarItem(placement: .topBarLeading) {
                    AccountMenuButton(currentUser: currentUser, isShowingDrawer: $isShowingDrawer)
                }
            }
            .accountDrawer(currentUser: currentUser, isPresented: $isShowingDrawer)
            .sheet(item: $selectedPeer) { peer in
                ChatView(currentUser: currentUser, peer: peer)
            }
        }
    }

    private func deleteConversation(at offsets: IndexSet) {
        for index in offsets {
            let peer = peers[index]
            for message in messages where (message.senderID == currentUser.id && message.receiverID == peer.id) || (message.senderID == peer.id && message.receiverID == currentUser.id) {
                modelContext.delete(message)
            }
        }
        try? modelContext.save()
    }
}

private struct ChatView: View {
    @Environment(\.modelContext) private var modelContext
    @Environment(\.dismiss) private var dismiss
    @Environment(\.appLanguage) private var appLanguage
    let currentUser: AppUser
    let peer: AppUser
    @Query(sort: \DirectMessage.createdAt) private var allMessages: [DirectMessage]
    @State private var text = ""
    @State private var isShowingProfile = false
    @State private var chatImageItem: PhotosPickerItem?
    @State private var chatVideoItem: PhotosPickerItem?

    private var messages: [DirectMessage] {
        allMessages.filter {
            ($0.senderID == currentUser.id && $0.receiverID == peer.id)
            || ($0.senderID == peer.id && $0.receiverID == currentUser.id)
        }
    }

    var body: some View {
        NavigationStack {
            ScrollViewReader { proxy in
                ScrollView {
                    LazyVStack(spacing: 10) {
                        Text(Date.now.formatted(date: .abbreviated, time: .omitted))
                            .font(.caption.weight(.semibold))
                            .foregroundStyle(.secondary)
                            .padding(.vertical, 8)
                        ForEach(messages) { message in
                            MessageBubble(message: message, isMine: message.senderID == currentUser.id)
                                .id(message.id)
                        }
                    }
                    .padding()
                }
                .background(KaiXTheme.pageBackground)
                .onAppear {
                    markRead()
                    proxy.scrollTo(messages.last?.id, anchor: .bottom)
                }
                .onChange(of: messages.count) {
                    proxy.scrollTo(messages.last?.id, anchor: .bottom)
                }
            }
            .safeAreaInset(edge: .bottom) {
                VStack(spacing: 8) {
                    HStack(spacing: 10) {
                        PhotosPicker(selection: $chatImageItem, matching: .images) {
                            Image(systemName: "photo")
                                .font(.headline)
                                .frame(width: 36, height: 36)
                                .background(Color(.secondarySystemBackground))
                                .clipShape(Circle())
                        }
                        .accessibilityLabel(L("sendImage", appLanguage))

                        PhotosPicker(selection: $chatVideoItem, matching: .videos) {
                            Image(systemName: "play.rectangle.fill")
                                .font(.headline)
                                .frame(width: 36, height: 36)
                                .background(Color(.secondarySystemBackground))
                                .clipShape(Circle())
                        }

                        TextField(L("sendMessage", appLanguage), text: $text)
                            .padding(.horizontal, 12)
                            .frame(height: 38)
                            .background(Color(.secondarySystemBackground))
                            .clipShape(Capsule())

                        Button {
                            sendMessage()
                        } label: {
                            Image(systemName: "arrow.up")
                                .font(.headline.weight(.bold))
                                .foregroundStyle(.white)
                                .frame(width: 38, height: 38)
                                .background(text.trimmingCharacters(in: .whitespacesAndNewlines).isEmpty ? Color.gray : Color.black)
                                .clipShape(Circle())
                        }
                        .disabled(text.trimmingCharacters(in: .whitespacesAndNewlines).isEmpty)
                    }
                }
                .padding(.horizontal)
                .padding(.vertical, 10)
                .background(.bar)
            }
            .navigationTitle(peer.displayName)
            .navigationBarTitleDisplayMode(.inline)
            .toolbar {
                ToolbarItem(placement: .cancellationAction) {
                    Button(L("close", appLanguage)) { dismiss() }
                }
                ToolbarItem(placement: .principal) {
                    HStack(spacing: 8) {
                        AvatarView(user: peer, size: 30)
                        VStack(alignment: .leading, spacing: 1) {
                            Text(peer.displayName)
                                .font(.subheadline.weight(.bold))
                            Text(L("online", appLanguage))
                                .font(.caption2)
                                .foregroundStyle(.green)
                        }
                    }
                }
                ToolbarItem(placement: .topBarTrailing) {
                    Button {
                        isShowingProfile = true
                    } label: {
                        Image(systemName: "person.crop.circle")
                    }
                    .accessibilityLabel(L("viewProfile", appLanguage))
                }
            }
            .sheet(isPresented: $isShowingProfile) {
                NavigationStack {
                    UserProfileView(currentUser: currentUser, user: peer)
                }
            }
            .onChange(of: chatImageItem) { _, newItem in
                Task { await sendPickedAttachment(newItem, kind: .image) }
            }
            .onChange(of: chatVideoItem) { _, newItem in
                Task { await sendPickedAttachment(newItem, kind: .video) }
            }
        }
    }

    private func sendMessage() {
        let trimmed = text.trimmingCharacters(in: .whitespacesAndNewlines)
        guard !trimmed.isEmpty else { return }
        modelContext.insert(DirectMessage(senderID: currentUser.id, receiverID: peer.id, content: trimmed))
        text = ""
        try? modelContext.save()
        Task {
            await SharedBackend.createMessage(senderID: currentUser.id, receiverID: peer.id, content: trimmed)
        }
    }

    @MainActor
    private func sendPickedAttachment(_ item: PhotosPickerItem?, kind: MessageAttachmentKind) async {
        guard let item, let data = try? await item.loadTransferable(type: Data.self) else { return }
        let content: String
        let symbol: String
        switch kind {
        case .video:
            content = videoSentText
            symbol = "play.rectangle.fill"
            chatVideoItem = nil
        case .image:
            content = L("sentImage", appLanguage)
            symbol = "photo"
            chatImageItem = nil
        case .none:
            return
        }
        modelContext.insert(DirectMessage(senderID: currentUser.id, receiverID: peer.id, content: content, attachmentSymbol: symbol, attachmentKind: kind, attachmentData: data))
        try? modelContext.save()
        Task {
            await SharedBackend.createMessage(senderID: currentUser.id, receiverID: peer.id, content: content, attachmentSymbol: symbol)
        }
    }

    private var videoSentText: String {
        L("videoSent", appLanguage)
    }

    private func markRead() {
        for message in messages where message.receiverID == currentUser.id && !message.isRead {
            message.isRead = true
        }
        try? modelContext.save()
    }
}

private struct AccountSecurityView: View {
    @Environment(\.modelContext) private var modelContext
    @Environment(\.appLanguage) private var appLanguage
    let currentUser: AppUser
    @Query(sort: \AppUser.joinedAt) private var users: [AppUser]
    @State private var handle: String
    @State private var email: String
    @State private var oldPassword = ""
    @State private var newPassword = ""
    @State private var message = ""
    @State private var isSuccess = false

    init(currentUser: AppUser) {
        self.currentUser = currentUser
        _handle = State(initialValue: currentUser.handle)
        _email = State(initialValue: currentUser.accountEmail)
    }

    var body: some View {
        Form {
            Section(L("accountInfo", appLanguage)) {
                TextField(L("username", appLanguage), text: $handle)
                    .textInputAutocapitalization(.never)
                    .autocorrectionDisabled()
                TextField(L("email", appLanguage), text: $email)
                    .textInputAutocapitalization(.never)
                    .keyboardType(.emailAddress)
            }

            Section(L("changePassword", appLanguage)) {
                SecureField(L("currentPassword", appLanguage), text: $oldPassword)
                SecureField(L("newPasswordMin", appLanguage), text: $newPassword)
            }

            if !message.isEmpty {
                Section {
                    Label(message, systemImage: isSuccess ? "checkmark.circle.fill" : "exclamationmark.circle.fill")
                        .foregroundStyle(isSuccess ? .green : .red)
                }
            }

            Section {
                Button(L("saveAccount", appLanguage)) {
                    save()
                }
                .fontWeight(.semibold)
            }
        }
        .navigationTitle(L("accountPassword", appLanguage))
    }

    private func save() {
        let normalizedHandle = handle.normalizedHandle
        message = ""
        isSuccess = false

        guard !normalizedHandle.isEmpty else {
            message = L("emptyHandle", appLanguage)
            return
        }

        guard users.allSatisfy({ $0.id == currentUser.id || $0.handle != normalizedHandle }) else {
            message = L("handleInUse", appLanguage)
            return
        }

        if !newPassword.isEmpty {
            guard oldPassword == currentUser.password else {
                message = L("wrongCurrentPassword", appLanguage)
                return
            }

            guard newPassword.count >= 6 else {
                message = L("newPasswordTooShort", appLanguage)
                return
            }

            currentUser.password = newPassword
            oldPassword = ""
            newPassword = ""
        }

        currentUser.handle = normalizedHandle
        currentUser.accountEmail = email.trimmingCharacters(in: .whitespacesAndNewlines)
        try? modelContext.save()
        isSuccess = true
        message = L("accountSaved", appLanguage)
        Task {
            await SharedBackend.updateUser(currentUser)
        }
    }
}

private enum ContentLibraryMode {
    case likes
    case bookmarks

    func title(_ language: AppLanguage) -> String {
        switch self {
        case .likes: L("likes", language)
        case .bookmarks: L("bookmarks", language)
        }
    }

    var kind: InteractionKind {
        switch self {
        case .likes: .like
        case .bookmarks: .bookmark
        }
    }

private struct ContentLibraryView: View {
    @Environment(\.appLanguage) private var appLanguage
    let currentUser: AppUser
    let mode: ContentLibraryMode
    @Query(sort: \FeedPost.createdAt, order: .reverse) private var posts: [FeedPost]
    @Query(sort: \AppUser.joinedAt) private var users: [AppUser]
    @Query private var interactions: [PostInteraction]
    @Query private var comments: [PostComment]

    private var savedPosts: [FeedPost] {
        let ids = Set(interactions.filter { $0.userID == currentUser.id && $0.kind == mode.kind.rawValue }.map(\.postID))
        return posts.filter { ids.contains($0.id) }
    }

    var body: some View {
        List {
            if savedPosts.isEmpty {
                ContentUnavailableView("\(L("emptyLibrary", appLanguage))\(mode.title(appLanguage))", systemImage: mode == .likes ? "heart" : "bookmark", description: Text(L("libraryHelp", appLanguage)))
            } else {
                ForEach(savedPosts) { post in
                    PostRow(post: post, currentUser: currentUser, users: users, interactions: interactions, comments: comments, heat: heat(for: post))
                        .listRowInsets(EdgeInsets())
                }
            }
        }
        .listStyle(.plain)
        .navigationTitle(mode.title(appLanguage))
    }

    private func heat(for post: FeedPost) -> Int {
        let likeCount = interactions.count { $0.postID == post.id && $0.kind == InteractionKind.like.rawValue }
        return post.viewCount + likeCount * 120
    }
}

private struct UserProfileView: View {
    @Environment(\.modelContext) private var modelContext
    @Environment(\.dismiss) private var dismiss
    @Environment(\.appLanguage) private var appLanguage
    let currentUser: AppUser
    let user: AppUser
    @Query(sort: \FeedPost.createdAt, order: .reverse) private var posts: [FeedPost]
    @Query(sort: \AppUser.joinedAt) private var users: [AppUser]
    @Query private var interactions: [PostInteraction]
    @Query private var comments: [PostComment]
    @Query private var follows: [FollowRelation]
    @State private var isShowingChat = false
    @State private var profileTab = ProfileContentTab.posts

    private var userPosts: [FeedPost] {
        posts.filter { $0.authorID == user.id }
    }

    private var visiblePosts: [FeedPost] {
        switch profileTab {
        case .posts:
            return userPosts
        case .replies:
            let repliedPostIDs = Set(comments.filter { $0.authorID == user.id }.map(\.postID))
            return posts.filter { repliedPostIDs.contains($0.id) }
        case .media:
            return userPosts.filter { $0.mediaKind != .none }
        case .likes:
            let likedPostIDs = Set(interactions.filter { $0.userID == user.id && $0.kind == InteractionKind.like.rawValue }.map(\.postID))
            return posts.filter { likedPostIDs.contains($0.id) }
        }
    }

    private var isFollowing: Bool {
        follows.contains { $0.followerID == currentUser.id && $0.followingID == user.id }
    }

    private var followerCount: Int {
        follows.filter { $0.followingID == user.id }.count
    }

    private var followingCount: Int {
        follows.filter { $0.followerID == user.id }.count
    }

    var body: some View {
        NavigationStack {
            VStack(spacing: 0) {
                StickyNavigationBar(title: user.displayName, onBack: { dismiss() })

                ScrollView {
                    VStack(alignment: .leading, spacing: 14) {
                        ProfileHeaderView(
                            user: user,
                            isCurrentUser: user.id == currentUser.id,
                            isFollowing: isFollowing,
                            postCount: userPosts.count,
                            replyCount: comments.filter { $0.authorID == user.id }.count,
                            likeCount: interactions.filter { $0.userID == user.id && $0.kind == InteractionKind.like.rawValue }.count,
                            bookmarkCount: interactions.filter { $0.userID == user.id && $0.kind == InteractionKind.bookmark.rawValue }.count,
                            followingCount: followingCount,
                            followerCount: followerCount,
                            onEdit: {},
                            onFollow: toggleFollow,
                            onMessage: { isShowingChat = true }
                        )

                        ProfileTabBar(selection: $profileTab)
                            .padding(.horizontal)

                        if visiblePosts.isEmpty {
                            ContentUnavailableView(emptyTitle, systemImage: emptyIcon)
                                .frame(maxWidth: .infinity)
                                .padding(.vertical, 34)
                        } else {
                            ForEach(visiblePosts) { post in
                                PostRow(post: post, currentUser: currentUser, users: users, interactions: interactions, comments: comments, heat: heat(for: post))
                            }
                        }
                    }
                }
                .safeAreaPadding(.bottom, KaiXTheme.bottomContentPadding)
            }
            .background(KaiXTheme.pageBackground)
            .toolbar(.hidden, for: .navigationBar)
            .sheet(isPresented: $isShowingChat) {
                ChatView(currentUser: currentUser, peer: user)
            }
        }
    }

    private var emptyTitle: String {
        switch profileTab {
        case .posts: L("noPostsFound", appLanguage)
        case .replies: L("noComments", appLanguage)
        case .media: L("noContent", appLanguage)
        case .likes: L("emptyLibrary", appLanguage) + L("likes", appLanguage)
        }
    }

    private var emptyIcon: String {
        switch profileTab {
        case .posts: "doc.text"
        case .replies: "bubble.left"
        case .media: "photo.on.rectangle"
        case .likes: "heart"
        }
    }

    private func toggleFollow() {
        if let existing = follows.first(where: { $0.followerID == currentUser.id && $0.followingID == user.id }) {
            modelContext.delete(existing)
        } else {
            modelContext.insert(FollowRelation(followerID: currentUser.id, followingID: user.id))
        }
        try? modelContext.save()
        Task {
            await SharedBackend.toggleFollow(followerID: currentUser.id, followingID: user.id)
        }
    }

    private func heat(for post: FeedPost) -> Int {
        let likeCount = interactions.count { $0.postID == post.id && $0.kind == InteractionKind.like.rawValue }
        let bookmarkCount = interactions.count { $0.postID == post.id && $0.kind == InteractionKind.bookmark.rawValue }
        return post.viewCount + likeCount * 120 + bookmarkCount * 40
    }
}

        _avatarImageData = State(initialValue: user.avatarImageData)

    var body: some View {

                    PhotosPicker(selection: $avatarPickerItem, matching: .images) {

                            HStack {
                                Circle()
                                    .fill(Color.named(color))
                                    .frame(width: 14, height: 14)

                    PhotosPicker(selection: $coverPickerItem, matching: .images) {

        user.location = location.trimmingCharacters(in: .whitespacesAndNewlines)
        user.websiteURL = websiteURL.trimmingCharacters(in: .whitespacesAndNewlines)
        user.avatarSymbol = avatarSymbol
        user.avatarColorName = avatarColorName

private struct ComposeView: View {
    @Environment(\.modelContext) private var modelContext
    @Environment(\.dismiss) private var dismiss
    @Environment(\.appLanguage) private var appLanguage
    let currentUser: AppUser
    @State private var text = ""
    @State private var selectedMediaKind: PostMediaKind = .none
    @State private var selectedImageItems: [PhotosPickerItem] = []
    @State private var selectedVideoItem: PhotosPickerItem?
    @State private var selectedImageData: [Data] = []
    @State private var selectedVideoData: Data?
    @State private var selectedVideoThumbnailData: Data?
    @State private var selectedVideoDuration = "00:30"
    @State private var isLoadingMedia = false
    @State private var showingSuccess = false
    @State private var isPublishing = false
    @State private var publishProgress: Double = 0
    @State private var publishError: String?
    @State private var isShowingDiscardDialog = false
    private let limit = 280

    private var remainingCount: Int {
        limit - text.count
    }

    private var canPost: Bool {
        (!text.trimmingCharacters(in: .whitespacesAndNewlines).isEmpty || !selectedImageData.isEmpty || selectedVideoData != nil) && remainingCount >= 0 && !isLoadingMedia && !isPublishing
    }

    private var hasDraft: Bool {
        !text.trimmingCharacters(in: .whitespacesAndNewlines).isEmpty || !selectedImageData.isEmpty || selectedVideoData != nil
    }

    var body: some View {
        NavigationStack {
            ZStack(alignment: .top) {
                Color(.systemGroupedBackground).ignoresSafeArea()

                VStack(spacing: 0) {
                    ComposeTopBar(
                        canPost: canPost,
                        isPublishing: isPublishing,
                        onCancel: requestCancel,
                        onPublish: publish
                    )

                    ScrollView {
                        VStack(alignment: .leading, spacing: 14) {
                            HStack(alignment: .top, spacing: 12) {
                                AvatarView(user: currentUser, size: 46)

                                VStack(alignment: .leading, spacing: 10) {
                                    ZStack(alignment: .topLeading) {
                                        TextEditor(text: $text)
                                            .font(.system(size: 18))
                                            .lineSpacing(4)
                                            .scrollContentBackground(.hidden)
                                            .frame(minHeight: 112, maxHeight: 210)
                                            .padding(.horizontal, 4)
                                            .padding(.vertical, 2)

                                        if text.isEmpty {
                                            Text(L("composePlaceholder", appLanguage))
                                                .font(.system(size: 18))
                                                .foregroundStyle(.secondary)
                                                .padding(.top, 10)
                                                .padding(.leading, 9)
                                                .allowsHitTesting(false)
                                        }
                                    }

                                    ScrollView(.horizontal, showsIndicators: false) {
                                        HStack(spacing: 8) {
                                            ForEach(localizedQuickTopics.prefix(6), id: \.self) { topic in
                                                Button {
                                                    appendTopic(topic)
                                                } label: {
                                                    Text(topic)
                                                        .font(.caption.weight(.bold))
                                                        .foregroundStyle(text.contains(topic) ? .white : .blue)
                                                        .padding(.horizontal, 10)
                                                        .padding(.vertical, 7)
                                                        .background(text.contains(topic) ? Color.blue : Color.blue.opacity(0.08))
                                                        .clipShape(Capsule())
                                                }
                                                .buttonStyle(.plain)
                                            }
                                        }
                                    }

                                    if !selectedImageData.isEmpty {
                                        ComposeImageGrid(imageData: selectedImageData, isUploading: isPublishing, progress: publishProgress, onDelete: removeImage)
                                    }

                                    if selectedVideoData != nil {
                                        ComposeVideoPreview(thumbnailData: selectedVideoThumbnailData, duration: selectedVideoDuration, isUploading: isPublishing, progress: publishProgress) {
                                            selectedVideoData = nil
                                            selectedVideoThumbnailData = nil
                                            selectedVideoDuration = "00:30"
                                            updateMediaKind()
                                        }
                                    }
                                }
                            }
                            .padding(.top, 4)
                            .padding(.horizontal, 2)

                            if let publishError {
                                HStack(spacing: 8) {
                                    Image(systemName: "exclamationmark.triangle.fill")
                                    Text(publishError)
                                    Spacer()
                                    Button(L("retry", appLanguage), action: publish)
                                        .font(.caption.weight(.bold))
                                }
                                .font(.caption)
                                .foregroundStyle(.red)
                                .padding(10)
                                .background(.red.opacity(0.08))
                                .clipShape(RoundedRectangle(cornerRadius: 12))
                            }
                        }
                        .padding(16)
                    }
                    .safeAreaPadding(.bottom, 72)

                    ComposeToolbar(
                        selectedImageItems: $selectedImageItems,
                        selectedVideoItem: $selectedVideoItem,
                        imageSlotsRemaining: max(0, 4 - selectedImageData.count),
                        canPickImages: selectedVideoData == nil && selectedImageData.count < 4 && !isPublishing,
                        canPickVideo: selectedImageData.isEmpty && !isPublishing,
                        countText: "\(text.count)/\(limit)",
                        remainingCount: remainingCount,
                        isLoadingMedia: isLoadingMedia || isPublishing,
                        onTopic: { appendTopic(defaultTopic) }
                    )
                }
                .onChange(of: selectedImageItems) { _, newItems in
                    Task { await loadImages(from: newItems) }
                }
                .onChange(of: selectedVideoItem) { _, newItem in
                    Task { await loadVideo(from: newItem) }
                }

                if showingSuccess {
                    Label(successText, systemImage: "checkmark.circle.fill")
                        .font(.headline)
                        .foregroundStyle(.white)
                        .padding(.horizontal, 16)
                        .padding(.vertical, 12)
                        .background(.black.opacity(0.86))
                        .clipShape(Capsule())
                        .padding(.top, 12)
                        .transition(.move(edge: .top).combined(with: .opacity))
                }
            }
            .toolbar(.hidden, for: .navigationBar)
            .confirmationDialog(L("discardDraftTitle", appLanguage), isPresented: $isShowingDiscardDialog, titleVisibility: .visible) {
                Button(L("keepEditing", appLanguage), role: .cancel) {}
                Button(L("discard", appLanguage), role: .destructive) {
                    dismiss()
                }
            }
        }
    }

    private var successText: String {
        L("postedSuccess", appLanguage)
    }

    private var defaultTopic: String {
        switch appLanguage {
        case .ja: "#話題"
        case .en: "#Topic"
        default: "#话题"
        }
    }

    private var localizedQuickTopics: [String] {
        switch appLanguage {
        case .ja: ["#KaiX", "#AI製品", "#SwiftUI", "#東京生活", "#開発ログ"]
        case .en: ["#KaiX", "#AIProducts", "#SwiftUI", "#TokyoLife", "#DevLog"]
        default: ["#KaiX", "#AI产品", "#SwiftUI", "#东京生活", "#开发日志"]
        }
    }

    private func requestCancel() {
        if hasDraft {
            isShowingDiscardDialog = true
        } else {
            dismiss()
        }
    }

    private func mediaLabel(_ kind: String) -> String {
        switch (appLanguage, kind) {
        case (.ja, "image"): "画像"
        case (.ja, "video"): "動画"
        case (.en, "image"): "Image"
        case (.en, "video"): "Video"
        case (_, "video"): "视频"
        default: "图片"
        }
    }

    @MainActor
    private func loadImages(from items: [PhotosPickerItem]) async {
        guard !items.isEmpty else { return }
        isLoadingMedia = true
        defer {
            isLoadingMedia = false
            selectedImageItems = []
            updateMediaKind()
        }
        for item in items.prefix(max(0, 4 - selectedImageData.count)) {
            if let data = try? await item.loadTransferable(type: Data.self), UIImage(data: data) != nil {
                selectedImageData.append(data)
            }
        }
        if selectedImageData.count > 4 {
            selectedImageData = Array(selectedImageData.prefix(4))
        }
        selectedVideoData = nil
    }

    @MainActor
    private func loadVideo(from item: PhotosPickerItem?) async {
        guard let item else { return }
        isLoadingMedia = true
        defer {
            isLoadingMedia = false
            selectedVideoItem = nil
            updateMediaKind()
        }
        if let data = try? await item.loadTransferable(type: Data.self) {
            selectedImageData = []
            selectedVideoData = data
            let metadata = await makeVideoMetadata(from: data)
            selectedVideoThumbnailData = metadata.thumbnailData
            selectedVideoDuration = metadata.duration
        }
    }

    private func makeVideoMetadata(from data: Data) async -> (thumbnailData: Data?, duration: String) {
        let url = FileManager.default.temporaryDirectory.appendingPathComponent("\(UUID().uuidString).mov")
        do {
            try data.write(to: url)
            let asset = AVURLAsset(url: url)
            let generator = AVAssetImageGenerator(asset: asset)
            generator.appliesPreferredTrackTransform = true
            generator.maximumSize = CGSize(width: 720, height: 0)
            let duration = (try? await asset.load(.duration)) ?? .zero
            let image = try await generator.image(at: CMTime(seconds: 0.2, preferredTimescale: 600)).image
            try? FileManager.default.removeItem(at: url)
            return (UIImage(cgImage: image).jpegData(compressionQuality: 0.78), duration.composeDurationText)
        } catch {
            try? FileManager.default.removeItem(at: url)
            return (nil, "00:30")
        }
    }

    private func removeImage(_ data: Data) {
        selectedImageData.removeAll { $0 == data }
        updateMediaKind()
    }

    private func updateMediaKind() {
        if selectedVideoData != nil {
            selectedMediaKind = .video
        } else if selectedImageData.count > 1 {
            selectedMediaKind = .gallery
        } else if selectedImageData.count == 1 {
            selectedMediaKind = .image
        } else {
            selectedMediaKind = .none
        }
    }

    private func appendTopic(_ topic: String) {
        if text.contains(topic) { return }
        if text.trimmingCharacters(in: .whitespacesAndNewlines).isEmpty {
            text = topic
        } else {
            text += " \(topic)"
        }
    }

    private func publish() {
        let trimmed = text.trimmingCharacters(in: .whitespacesAndNewlines)
        updateMediaKind()
        guard !trimmed.isEmpty || selectedMediaKind != .none else { return }
        guard remainingCount >= 0 else { return }
        isPublishing = true
        publishProgress = 0.28
        publishError = nil
        let content = trimmed.isEmpty ? "#KaiX" : trimmed
        modelContext.insert(FeedPost(
            authorID: currentUser.id,
            content: content,
            viewCount: 1,
            mediaKind: selectedMediaKind,
            mediaSymbol: selectedMediaKind == .video ? "play.rectangle.fill" : "photo.fill",
            mediaColorName: currentUser.avatarColorName,
            mediaTitle: selectedMediaKind == .video ? mediaLabel("video") : mediaLabel("image"),
            videoDuration: selectedMediaKind == .video ? selectedVideoDuration : "",
            mediaImageData: selectedMediaKind == .video ? Array([selectedVideoThumbnailData].compactMap { $0 }.prefix(1)) : selectedImageData,
            videoData: selectedVideoData
        ))
        publishProgress = 0.74
        guard (try? modelContext.save()) != nil else {
            publishError = L("failedToPost", appLanguage)
            isPublishing = false
            return
        }
        Task {
            await SharedBackend.createPost(userID: currentUser.id, content: content)
        }
        publishProgress = 1
        withAnimation(.snappy) {
            showingSuccess = true
        }
        DispatchQueue.main.asyncAfter(deadline: .now() + 0.7) {
            isPublishing = false
            dismiss()
        }
    }
}

private extension FeedPost {
    var isRising: Bool {
        Date.now.timeIntervalSince(createdAt) < 1800
    }

    var mediaKind: PostMediaKind {
        get { PostMediaKind(rawValue: mediaKindRaw) ?? .none }
        set { mediaKindRaw = newValue.rawValue }
    }

    var imageDataItems: [Data] {
        [mediaImageData1, mediaImageData2, mediaImageData3, mediaImageData4].compactMap { $0 }
    }

    var tags: [String] {
        content
            .split(separator: " ")
            .map(String.init)
            .filter { $0.hasPrefix("#") && $0.count > 1 }
    }

    var bodyWithoutTrailingTags: String {
        let parts = content.split(separator: " ").map(String.init)
        let trimmed = parts.dropLast(parts.reversed().prefix { $0.hasPrefix("#") }.count)
        let result = trimmed.joined(separator: " ")
        return result.isEmpty ? content : result
    }
}

private extension DirectMessage {
    var attachmentKind: MessageAttachmentKind {
        get { MessageAttachmentKind(rawValue: attachmentKindRaw) ?? .none }
        set { attachmentKindRaw = newValue.rawValue }
    }
}

private extension Color {
    static func named(_ name: String) -> Color {
        switch name {
        case "blue": .blue
        case "green": .green
        case "orange": .orange
        case "pink": .pink
        case "purple": .purple
        case "indigo": .indigo
        case "teal": .teal
        case "red": .red
        case "yellow": .yellow
        case "cyan": .cyan
        case "mint": .mint
        case "brown": .brown
        default: .black
        }
    }
}

private extension String {
    var normalizedHandle: String {
        trimmingCharacters(in: .whitespacesAndNewlines)
            .replacingOccurrences(of: "@", with: "")
            .lowercased()
    }

    var headlineText: String {
        let clean = replacingOccurrences(of: "\n", with: " ")
        if clean.count <= 34 {
            return clean
        }
        return "\(clean.prefix(34))..."
    }
}

private extension Date {
    var relativeText: String {
        let formatter = RelativeDateTimeFormatter()
        let code = UserDefaults.standard.string(forKey: "appLanguageCode") ?? AppLanguage.system.rawValue
        switch AppLanguage.resolved(from: code) {
        case .ja:
            formatter.locale = Locale(identifier: "ja_JP")
        case .en:
            formatter.locale = Locale(identifier: "en_US")
        default:
            formatter.locale = Locale(identifier: "zh_Hans")
        }
        formatter.unitsStyle = .short
        return formatter.localizedString(for: self, relativeTo: .now)
    }
}
